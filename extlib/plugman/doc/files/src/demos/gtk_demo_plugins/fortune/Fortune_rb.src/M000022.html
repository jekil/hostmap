  def display()
    @logger.debug { "Fortune displaying" }
    box = Gtk::VBox.new
    label = Gtk::Label.new(`fortune`)
    box.pack_start(label, true, true, 0)
    button = Gtk::Button.new("Another?")  
    box.pack_start(button, false, false, 0)
    
    button.signal_connect("clicked") {
      label.text = `fortune`
    }
    box
  end
