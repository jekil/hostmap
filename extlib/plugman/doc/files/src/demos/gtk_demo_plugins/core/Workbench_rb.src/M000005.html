  def make_tab_widget(text, img_loc)
    tab_widget = Gtk::HBox.new
    image = Gtk::Image.new(img_loc)
    label = Gtk::Label.new(text, true)
    tab_widget.pack_start(image)
    tab_widget.pack_start(label)
    tab_widget.resize_children
    tab_widget.show_all
    tab_widget
  end
